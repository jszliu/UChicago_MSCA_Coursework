def train(clf_name, X, y):
    """ train model """
    clf = classifiers[clf_name]
    y_hat = cross_val_predict(clf, X, y, cv=10)
    clf.fit(X, y)
    return clf, y_hat


def evaluate(y, y_hat):
    """ evaluate model performance """
    metrics = []
    for metric in [accuracy_score, precision_score, recall_score, f1_score]:
        metrics.append(metric(y, y_hat))
    return metrics


def plot_metric(title, name, metric_index, metrics):
    clfs = list(metrics.keys())
    values = [v[metric_index] for v in metrics.values()]
    indices = list(map(int, np.argsort(values)[::-1]))
    clfs = [clfs[i] for i in indices]
    values = [values[i] for i in indices]
    plt.figure(figsize=(10, 6))
    plt.bar(clfs, values)
    plt.title('%s - %s' % (title, name))
    plt.ylabel(name)
    plt.ylim(min(values) - 0.1, 1.)
    plt.xticks(rotation=45)


def format_metrics(metrics):
    return ', '.join(['%.4f' % m for m in metrics])


def save_model(clf, clf_name):
    joblib.dump(clf, clf_name + '.joblib')

def load_model(clf_name):
    return joblib.load(clf_name + '.joblib')

all_train_metrics, all_test_metrics = {}, {}
for clf_name in classifiers:
    start = time.time()
    clf, y_cv_hat = train(clf_name, X_train, y_train)
    save_model(clf, clf_name)
    cost = 1000 * (time.time() - start)
    train_metrics = evaluate(y_train, y_cv_hat)
    test_metrics = evaluate(y_test, clf.predict(X_test))
    print('classifier[%s] train_metrics[%s] test_metrics[%s] time_ms[%d]'
          % (clf_name, format_metrics(train_metrics), format_metrics(test_metrics), cost))
    all_train_metrics[clf_name] = train_metrics
    all_test_metrics[clf_name] = test_metrics

plot_metric('10-fold-cross-validation performance', 'Accuracy', 0, all_train_metrics)
plot_metric('10-fold-cross-validation performance', 'Precision', 1, all_train_metrics)
plot_metric('10-fold-cross-validation performance', 'Recall', 2, all_train_metrics)
plot_metric('10-fold-cross-validation performance', 'F1-score', 3, all_train_metrics)
plot_metric('The performance on the validation set', 'Accuracy', 0, all_test_metrics)
plot_metric('The performance on the validation set', 'Precision', 1, all_test_metrics)
plot_metric('The performance on the validation set', 'Recall', 2, all_test_metrics)

In this study, we try various machine learning method for the classification task, including LogisticRegression, KNN, DecisionTree, RandomForest, GaussianNB, SVM, Perceptron, MLP, SGD and GradientBoosting. Most of them are simple models with nonlinear transformation for features. RandomForest and GradientBoosting are ensemble learning methods which consist of several simple classifiers. Simple classifiers are fast to train and predict, while ensemble learning methods cost more time.
