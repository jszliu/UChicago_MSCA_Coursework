def qry(sql):

    # Open database connection
    connection = pymysql.connect(host='localhost',
                             user='root',
                             password='rootroot',
                             database='sakila',
                             charset='utf8mb4',
                             cursorclass=pymysql.cursors.DictCursor)
    df = pd.read_sql(sql, connection)

    # disconnect from server
    connection.close()
    
    # return data.
    return df
